43 :type client: :class:`gcloud.storage.client.Client` 

44 :param client: Optional. The client to use for making connections. 

45 Defaults to the bucket's client. 

46 """ 

47 def __init__(self, bucket, extra_params=None, client=None): 

48 if client is None:

57 def get_items_from_response(self, response): 

58 """Yield :class:`.storage.blob.Blob` items from response. 

59 

60 :type response: dict

126 def exists(self, client=None): 

127 """Determines whether or not this bucket exists. 

128 

129 :type client: :class:`gcloud.storage.client.Client` or ``NoneType`` 

130 :param client: Optional. The client to use. If not passed, falls back 

131 to the ``client`` stored on the current bucket. 

132 

133 :rtype: boolean 

134 :returns: True if the bucket exists in Cloud Storage. 

135 """ 

136 client = self._require_client(client) 

137 try: 

138 # We only need the status code (200 or not) so we seek to 

139 # minimize the returned payload. 

140 query_params = {'fields': 'name'} 

141 # We intentionally pass `_target_object=None` since fields=name 

142 # would limit the local properties. 

143 client.connection.api_request(method='GET', path=self.path, 

144 query_params=query_params, 

145 _target_object=None) 

146 # NOTE: This will not fail immediately in a batch. However, when 

147 # Batch.finish() is called, the resulting `NotFound` will be 

148 # raised. 

149 return True 

150 except NotFound: 

151 return False

154 """Creates current bucket. 

155 

156 If the bucket already exists, will raise

158 

159 This implements "storage.buckets.insert". 

160 

161 :type client: :class:`gcloud.storage.client.Client` or ``NoneType`` 

162 :param client: Optional. The client to use. If not passed, falls back 

163 to the ``client`` stored on the current bucket. 

164 """ 

165 client = self._require_client(client) 

166 query_params = {'project': client.project} 

167 properties = dict( 

168 (key, self._properties[key]) for key in self._changes) 

169 properties['name'] = self.name 

170 api_response = client.connection.api_request( 

171 method='POST', path='/b', query_params=query_params, 

172 data=properties, _target_object=self) 

173 self._set_properties(api_response)

205 def get_blob(self, blob_name, client=None): 

206 """Get a blob object by name. 

207 

208 This will return None if the blob doesn't exist:: 

209 

210 >>> from gcloud import storage 

211 >>> client = storage.Client() 

212 >>> bucket = client.get_bucket('my-bucket') 

213 >>> print bucket.get_blob('/path/to/blob.txt') 

214 <Blob: my-bucket, /path/to/blob.txt> 

215 >>> print bucket.get_blob('/does-not-exist.txt') 

216 None 

217 

218 :type blob_name: string 

219 :param blob_name: The name of the blob to retrieve. 

220 

221 :type client: :class:`gcloud.storage.client.Client` or ``NoneType`` 

222 :param client: Optional. The client to use. If not passed, falls back 

223 to the ``client`` stored on the current bucket. 

224 

225 :rtype: :class:`gcloud.storage.blob.Blob` or None 

226 :returns: The blob object if it exists, otherwise None. 

227 """ 

228 client = self._require_client(client) 

229 blob = Blob(bucket=self, name=blob_name) 

230 try: 

231 response = client.connection.api_request( 

232 method='GET', path=blob.path, _target_object=blob) 

233 # NOTE: We assume response.get('name') matches `blob_name`. 

234 blob._set_properties(response) 

235 # NOTE: This will not fail immediately in a batch. However, when 

236 # Batch.finish() is called, the resulting `NotFound` will be 

237 # raised. 

238 return blob 

239 except NotFound: 

240 return None

242 def list_blobs(self, max_results=None, page_token=None, prefix=None, 

243 delimiter=None, versions=None, 

244 projection='noAcl', fields=None, client=None): 

245 """Return an iterator used to find blobs in the bucket. 

246 

247 :type max_results: integer or ``NoneType`` 

248 :param max_results: maximum number of blobs to return. 

249 

250 :type page_token: string 

251 :param page_token: opaque marker for the next "page" of blobs. If not 

252 passed, will return the first page of blobs. 

253 

254 :type prefix: string or ``NoneType`` 

255 :param prefix: optional prefix used to filter blobs. 

256 

257 :type delimiter: string or ``NoneType`` 

258 :param delimiter: optional delimter, used with ``prefix`` to 

259 emulate hierarchy. 

260 

261 :type versions: boolean or ``NoneType`` 

262 :param versions: whether object versions should be returned as 

263 separate blobs. 

264 

265 :type projection: string or ``NoneType`` 

266 :param projection: If used, must be 'full' or 'noAcl'. Defaults to 

267 'noAcl'. Specifies the set of properties to return. 

268 

269 :type fields: string or ``NoneType`` 

270 :param fields: Selector specifying which fields to include in a 

271 partial response. Must be a list of fields. For example 

272 to get a partial response with just the next page token 

273 and the language of each blob returned: 

274 'items/contentLanguage,nextPageToken' 

275 

276 :type client: :class:`gcloud.storage.client.Client` or ``NoneType`` 

277 :param client: Optional. The client to use. If not passed, falls back 

278 to the ``client`` stored on the current bucket. 

279 

280 :rtype: :class:`_BlobIterator`. 

281 :returns: An iterator of blobs. 

282 """ 

283 extra_params = {} 

284 

285 if max_results is not None: 

286 extra_params['maxResults'] = max_results 

287 

288 if prefix is not None: 

289 extra_params['prefix'] = prefix 

290 

291 if delimiter is not None: 

292 extra_params['delimiter'] = delimiter 

293 

294 if versions is not None: 

295 extra_params['versions'] = versions 

296 

297 extra_params['projection'] = projection 

298 

299 if fields is not None: 

300 extra_params['fields'] = fields 

301 

302 result = self._iterator_class( 

303 self, extra_params=extra_params, client=client) 

304 # Page token must be handled specially since the base `Iterator` 

305 # class has it as a reserved property. 

306 if page_token is not None: 

307 result.next_page_token = page_token 

308 return result

310 def delete(self, force=False, client=None): 

311 """Delete this bucket. 

312 

313 The bucket **must** be empty in order to submit a delete request. If 

314 ``force=True`` is passed, this will first attempt to delete all the 

315 objects / blobs in the bucket (i.e. try to empty the bucket). 

316 

317 If the bucket doesn't exist, this will raise 

318 :class:`gcloud.exceptions.NotFound`. If the bucket is not empty 

319 (and ``force=False``), will raise :class:`gcloud.exceptions.Conflict`. 

320 

321 If ``force=True`` and the bucket contains more than 256 objects / blobs 

322 this will cowardly refuse to delete the objects (or the bucket). This 

323 is to prevent accidental bucket deletion and to prevent extremely long 

324 runtime of this method. 

325 

326 :type force: boolean 

327 :param force: If True, empties the bucket's objects then deletes it. 

328 

329 :type client: :class:`gcloud.storage.client.Client` or ``NoneType`` 

330 :param client: Optional. The client to use. If not passed, falls back 

331 to the ``client`` stored on the current bucket. 

332 

333 :raises: :class:`ValueError` if ``force`` is ``True`` and the bucket 

334 contains more than 256 objects / blobs. 

335 """ 

336 client = self._require_client(client) 

337 if force: 

338 blobs = list(self.list_blobs( 

339 max_results=self._MAX_OBJECTS_FOR_ITERATION + 1, 

340 client=client)) 

341 if len(blobs) > self._MAX_OBJECTS_FOR_ITERATION: 

342 message = ( 

343 'Refusing to delete bucket with more than ' 

344 '%d objects. If you actually want to delete ' 

345 'this bucket, please delete the objects ' 

346 'yourself before calling Bucket.delete().' 

347 ) % (self._MAX_OBJECTS_FOR_ITERATION,) 

348 raise ValueError(message) 

349 

350 # Ignore 404 errors on delete. 

351 self.delete_blobs(blobs, on_error=lambda blob: None, 

352 client=client) 

353 

354 # We intentionally pass `_target_object=None` since a DELETE 

355 # request has no response value (whether in a standard request or 

356 # in a batch request). 

357 client.connection.api_request(method='DELETE', path=self.path, 

358 _target_object=None)

360 def delete_blob(self, blob_name, client=None): 

361 """Deletes a blob from the current bucket. 

362 

363 If the blob isn't found (backend 404), raises a 

364 :class:`gcloud.exceptions.NotFound`. 

365 

366 For example:: 

367 

368 >>> from gcloud.exceptions import NotFound 

369 >>> from gcloud import storage 

370 >>> client = storage.Client() 

371 >>> bucket = client.get_bucket('my-bucket') 

372 >>> print bucket.list_blobs() 

373 [<Blob: my-bucket, my-file.txt>] 

374 >>> bucket.delete_blob('my-file.txt') 

375 >>> try: 

376 ... bucket.delete_blob('doesnt-exist') 

377 ... except NotFound: 

378 ... pass 

379 

380 :type blob_name: string 

381 :param blob_name: A blob name to delete. 

382 

383 :type client: :class:`gcloud.storage.client.Client` or ``NoneType`` 

384 :param client: Optional. The client to use. If not passed, falls back 

385 to the ``client`` stored on the current bucket. 

386 

387 :raises: :class:`gcloud.exceptions.NotFound` (to suppress 

388 the exception, call ``delete_blobs``, passing a no-op 

389 ``on_error`` callback, e.g.:: 

390 

391 >>> bucket.delete_blobs([blob], on_error=lambda blob: None) 

392 """ 

393 client = self._require_client(client) 

394 blob_path = Blob.path_helper(self.path, blob_name) 

395 # We intentionally pass `_target_object=None` since a DELETE 

396 # request has no response value (whether in a standard request or 

397 # in a batch request). 

398 client.connection.api_request(method='DELETE', path=blob_path, 

399 _target_object=None)

401 def delete_blobs(self, blobs, on_error=None, client=None): 

402 """Deletes a list of blobs from the current bucket. 

403 

404 Uses :func:`Bucket.delete_blob` to delete each individual blob. 

405 

406 :type blobs: list of string or :class:`gcloud.storage.blob.Blob` 

407 :param blobs: A list of blob names or Blob objects to delete. 

408 

409 :type on_error: a callable taking (blob) 

410 :param on_error: If not ``None``, called once for each blob raising 

411 :class:`gcloud.exceptions.NotFound`; 

412 otherwise, the exception is propagated. 

413 

414 :type client: :class:`gcloud.storage.client.Client` or ``NoneType`` 

415 :param client: Optional. The client to use. If not passed, falls back 

416 to the ``client`` stored on the current bucket. 

417 

418 :raises: :class:`gcloud.exceptions.NotFound` (if 

419 `on_error` is not passed). 

420 """ 

421 for blob in blobs: 

422 try: 

423 blob_name = blob 

424 if not isinstance(blob_name, six.string_types): 

425 blob_name = blob.name 

426 self.delete_blob(blob_name, client=client) 

427 except NotFound: 

428 if on_error is not None: 

429 on_error(blob) 

430 else: 

431 raise

464 def rename_blob(self, blob, new_name, client=None): 

465 """Rename the given blob using copy and delete operations. 

466 

467 Effectively, copies blob to the same bucket with a new name, then 

468 deletes the blob. 

469 

470 .. warning:: 

471 

472 This method will first duplicate the data and then delete the 

473 old blob. This means that with very large objects renaming 

474 could be a very (temporarily) costly or a very slow operation. 

475 

476 :type blob: :class:`gcloud.storage.blob.Blob` 

477 :param blob: The blob to be renamed. 

478 

479 :type new_name: string 

480 :param new_name: The new name for this blob. 

481 

482 :type client: :class:`gcloud.storage.client.Client` or ``NoneType`` 

483 :param client: Optional. The client to use. If not passed, falls back 

484 to the ``client`` stored on the current bucket. 

485 

486 :rtype: :class:`Blob` 

487 :returns: The newly-renamed blob. 

488 """ 

489 new_blob = self.copy_blob(blob, self, new_name, client=client) 

490 blob.delete(client=client) 

491 return new_blob

773 def make_public(self, recursive=False, future=False, client=None): 

774 """Make a bucket public. 

775 

776 If ``recursive=True`` and the bucket contains more than 256 

777 objects / blobs this will cowardly refuse to make the objects public. 

778 This is to prevent extremely long runtime of this method. 

779 

780 :type recursive: boolean 

781 :param recursive: If True, this will make all blobs inside the bucket 

782 public as well. 

783 

784 :type future: boolean 

785 :param future: If True, this will make all objects created in the 

786 future public as well. 

787 

788 :type client: :class:`gcloud.storage.client.Client` or ``NoneType`` 

789 :param client: Optional. The client to use. If not passed, falls back 

790 to the ``client`` stored on the current bucket. 

791 """ 

792 self.acl.all().grant_read() 

793 self.acl.save(client=client) 

794 

795 if future: 

796 doa = self.default_object_acl 

797 if not doa.loaded: 

798 doa.reload(client=client) 

799 doa.all().grant_read() 

800 doa.save(client=client) 

801 

802 if recursive: 

803 blobs = list(self.list_blobs( 

804 projection='full', 

805 max_results=self._MAX_OBJECTS_FOR_ITERATION + 1, 

806 client=client)) 

807 if len(blobs) > self._MAX_OBJECTS_FOR_ITERATION: 

808 message = ( 

809 'Refusing to make public recursively with more than ' 

810 '%d objects. If you actually want to make every object ' 

811 'in this bucket public, please do it on the objects ' 

812 'yourself.' 

813 ) % (self._MAX_OBJECTS_FOR_ITERATION,) 

814 raise ValueError(message) 

815 

816 for blob in blobs: 

817 blob.acl.all().grant_read() 

818 blob.acl.save(client=client)
