15"""Manipulate access control lists that Cloud Storage provides. 

16 

17:class:`gcloud.storage.bucket.Bucket` has a getting method that creates 

18an ACL object under the hood, and you can interact with that using 

19:func:`gcloud.storage.bucket.Bucket.acl`:: 

20 

21 >>> from gcloud import storage 

22 >>> client = storage.Client() 

23 >>> bucket = client.get_bucket(bucket_name) 

24 >>> acl = bucket.acl 

25 

26Adding and removing permissions can be done with the following methods 

27(in increasing order of granularity): 

28 

29- :func:`ACL.all` 

30 corresponds to access for all users. 

31- :func:`ACL.all_authenticated` corresponds 

32 to access for all users that are signed into a Google account. 

33- :func:`ACL.domain` corresponds to access on a 

34 per Google Apps domain (ie, ``example.com``). 

35- :func:`ACL.group` corresponds to access on a 

36 per group basis (either by ID or e-mail address). 

37- :func:`ACL.user` corresponds to access on a 

38 per user basis (either by ID or e-mail address). 

39 

40And you are able to ``grant`` and ``revoke`` the following roles: 

41 

42- **Reading**: 

43 :func:`_ACLEntity.grant_read` and :func:`_ACLEntity.revoke_read` 

44- **Writing**: 

45 :func:`_ACLEntity.grant_write` and :func:`_ACLEntity.revoke_write` 

46- **Owning**: 

47 :func:`_ACLEntity.grant_owner` and :func:`_ACLEntity.revoke_owner` 

48 

49You can use any of these like any other factory method (these happen to 

50be :class:`_ACLEntity` factories):: 

51 

52 >>> acl.user('me@example.org').grant_read() 

53 >>> acl.all_authenticated().grant_write() 

54 

55You can also chain these ``grant_*`` and ``revoke_*`` methods together 

56for brevity:: 

57 

58 >>> acl.all().grant_read().revoke_write() 

59 

60After that, you can save any changes you make with the 

61:func:`gcloud.storage.acl.ACL.save` method:: 

62 

63 >>> acl.save() 

64 

65You can alternatively save any existing :class:`gcloud.storage.acl.ACL` 

66object (whether it was created by a factory method or not) from a 

67:class:`gcloud.storage.bucket.Bucket`:: 

68 

69 >>> bucket.acl.save(acl=acl) 

70 

71To get the list of ``entity`` and ``role`` for each unique pair, the 

72:class:`ACL` class is iterable:: 

73 

74 >>> print list(ACL) 

75 [{'role': 'OWNER', 'entity': 'allUsers'}, ...] 

76 

77This list of tuples can be used as the ``entity`` and ``role`` fields 

78when sending metadata for ACLs to the API. 

79"""

82class _ACLEntity(object): 

83 """Class representing a set of roles for an entity. 

84 

85 This is a helper class that you likely won't ever construct 

86 outside of using the factor methods on the :class:`ACL` object. 

87 

88 :type entity_type: string 

89 :param entity_type: The type of entity (ie, 'group' or 'user'). 

90 

91 :type identifier: string 

92 :param identifier: The ID or e-mail of the entity. For the special 

93 entity types (like 'allUsers') this is optional. 

94 """ 

95 

96 READER_ROLE = 'READER' 

97 WRITER_ROLE = 'WRITER' 

98 OWNER_ROLE = 'OWNER' 

99 

100 def __init__(self, entity_type, identifier=None): 

101 self.identifier = identifier 

102 self.roles = set([]) 

103 self.type = entity_type 

104 

105 def __str__(self): 

106 if not self.identifier: 

107 return str(self.type) 

108 else: 

109 return '{acl.type}-{acl.identifier}'.format(acl=self) 

110 

111 def __repr__(self): 

112 return '<ACL Entity: {acl} ({roles})>'.format( 

113 acl=self, roles=', '.join(self.roles)) 

114 

115 def get_roles(self): 

116 """Get the list of roles permitted by this entity. 

117 

118 :rtype: list of strings 

119 :returns: The list of roles associated with this entity. 

120 """ 

121 return self.roles 

122 

123 def grant(self, role): 

124 """Add a role to the entity. 

125 

126 :type role: string 

127 :param role: The role to add to the entity. 

128 """ 

129 self.roles.add(role) 

130 

131 def revoke(self, role): 

132 """Remove a role from the entity. 

133 

134 :type role: string 

135 :param role: The role to remove from the entity. 

136 """ 

137 if role in self.roles: 

138 self.roles.remove(role)

222 def entity_from_dict(self, entity_dict): 

223 """Build an _ACLEntity object from a dictionary of data. 

224 

225 An entity is a mutable object that represents a list of roles 

226 belonging to either a user or group or the special types for all 

227 users and all authenticated users. 

228 

229 :type entity_dict: dict 

230 :param entity_dict: Dictionary full of data from an ACL lookup. 

231 

232 :rtype: :class:`_ACLEntity` 

233 :returns: An Entity constructed from the dictionary. 

234 """ 

235 entity = entity_dict['entity'] 

236 role = entity_dict['role'] 

237 

238 if entity == 'allUsers': 

239 entity = self.all() 

240 

241 elif entity == 'allAuthenticatedUsers': 

242 entity = self.all_authenticated() 

243 

244 elif '-' in entity: 

245 entity_type, identifier = entity.split('-', 1) 

246 entity = self.entity(entity_type=entity_type, 

247 identifier=identifier) 

248 

249 if not isinstance(entity, _ACLEntity): 

250 raise ValueError('Invalid dictionary: %s' % entity_dict) 

251 

252 entity.grant(role) 

253 return entity

293 def entity(self, entity_type, identifier=None): 

294 """Factory method for creating an Entity. 

295 

296 If an entity with the same type and identifier already exists, 

297 this will return a reference to that entity. If not, it will 

298 create a new one and add it to the list of known entities for 

299 this ACL. 

300 

301 :type entity_type: string 

302 :param entity_type: The type of entity to create 

303 (ie, ``user``, ``group``, etc) 

304 

305 :type identifier: string 

306 :param identifier: The ID of the entity (if applicable). 

307 This can be either an ID or an e-mail address. 

308 

309 :rtype: :class:`_ACLEntity` 

310 :returns: A new Entity or a reference to an existing identical entity.

312 entity = _ACLEntity(entity_type=entity_type, identifier=identifier) 

313 if self.has_entity(entity): 

314 entity = self.get_entity(entity)

396 def reload(self, client=None): 

397 """Reload the ACL data from Cloud Storage. 

398 

399 :type client: :class:`gcloud.storage.client.Client` or ``NoneType`` 

400 :param client: Optional. The client to use. If not passed, falls back 

401 to the ``client`` stored on the ACL's parent. 

402 """ 

403 path = self.reload_path 

404 client = self._require_client(client) 

405 

406 self.entities.clear() 

407 

408 found = client.connection.api_request(method='GET', path=path) 

409 self.loaded = True 

410 for entry in found.get('items', ()): 

411 self.add_entity(self.entity_from_dict(entry))

413 def _save(self, acl, predefined, client): 

414 """Helper for :meth:`save` and :meth:`save_predefined`. 

415 

416 :type acl: :class:`gcloud.storage.acl.ACL`, or a compatible list. 

417 :param acl: The ACL object to save. If left blank, this will save 

418 current entries. 

419 

420 :type predefined: string or None 

421 :param predefined: An identifier for a predefined ACL. Must be one 

422 of the keys in :attr:`PREDEFINED_JSON_ACLS` 

423 If passed, `acl` must be None. 

424 

425 :type client: :class:`gcloud.storage.client.Client` or ``NoneType`` 

426 :param client: Optional. The client to use. If not passed, falls back 

427 to the ``client`` stored on the ACL's parent. 

428 """ 

429 query_params = {'projection': 'full'} 

430 if predefined is not None: 

431 acl = [] 

432 query_params[self._PREDEFINED_QUERY_PARAM] = predefined 

433 

434 path = self.save_path 

435 client = self._require_client(client) 

436 result = client.connection.api_request( 

437 method='PATCH', 

438 path=path, 

439 data={self._URL_PATH_ELEM: list(acl)}, 

440 query_params=query_params) 

441 self.entities.clear() 

442 for entry in result.get(self._URL_PATH_ELEM, ()): 

443 self.add_entity(self.entity_from_dict(entry)) 

444 self.loaded = True

446 def save(self, acl=None, client=None): 

447 """Save this ACL for the current bucket. 

448 

449 :type acl: :class:`gcloud.storage.acl.ACL`, or a compatible list. 

450 :param acl: The ACL object to save. If left blank, this will save 

451 current entries. 

452 

453 :type client: :class:`gcloud.storage.client.Client` or ``NoneType`` 

454 :param client: Optional. The client to use. If not passed, falls back 

455 to the ``client`` stored on the ACL's parent. 

456 """ 

457 if acl is None: 

458 acl = self 

459 save_to_backend = acl.loaded 

460 else: 

461 save_to_backend = True 

462 

463 if save_to_backend: 

464 self._save(acl, None, client)

466 def save_predefined(self, predefined, client=None): 

467 """Save this ACL for the current bucket using a predefined ACL. 

468 

469 :type predefined: string 

470 :param predefined: An identifier for a predefined ACL. Must be one 

471 of the keys in :attr:`PREDEFINED_JSON_ACLS` 

472 or :attr:`PREDEFINED_XML_ACLS` (which will be 

473 aliased to the corresponding JSON name). 

474 If passed, `acl` must be None. 

475 

476 :type client: :class:`gcloud.storage.client.Client` or ``NoneType`` 

477 :param client: Optional. The client to use. If not passed, falls back 

478 to the ``client`` stored on the ACL's parent. 

479 """ 

480 predefined = self.PREDEFINED_XML_ACLS.get(predefined, predefined) 

481 

482 if predefined not in self.PREDEFINED_JSON_ACLS: 

483 raise ValueError("Invalid predefined ACL: %s" % (predefined,)) 

484 

485 self._save(None, predefined, client)
