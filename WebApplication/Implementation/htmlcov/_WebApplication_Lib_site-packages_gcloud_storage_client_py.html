144 def get_bucket(self, bucket_name): 

145 """Get a bucket by name. 

146 

147 If the bucket isn't found, this will raise a 

148 :class:`gcloud.storage.exceptions.NotFound`. 

149 

150 For example:: 

151 

152 >>> try: 

153 >>> bucket = client.get_bucket('my-bucket') 

154 >>> except gcloud.exceptions.NotFound: 

155 >>> print 'Sorry, that bucket does not exist!' 

156 

157 This implements "storage.buckets.get". 

158 

159 :type bucket_name: string 

160 :param bucket_name: The name of the bucket to get. 

161 

162 :rtype: :class:`gcloud.storage.bucket.Bucket` 

163 :returns: The bucket matching the name provided. 

164 :raises: :class:`gcloud.exceptions.NotFound` 

165 """ 

166 bucket = Bucket(self, name=bucket_name) 

167 bucket.reload(client=self) 

168 return bucket

170 def lookup_bucket(self, bucket_name): 

171 """Get a bucket by name, returning None if not found. 

172 

173 You can use this if you would rather check for a None value 

174 than catching an exception:: 

175 

176 >>> bucket = client.lookup_bucket('doesnt-exist') 

177 >>> print bucket 

178 None 

179 >>> bucket = client.lookup_bucket('my-bucket') 

180 >>> print bucket 

181 <Bucket: my-bucket> 

182 

183 :type bucket_name: string 

184 :param bucket_name: The name of the bucket to get. 

185 

186 :rtype: :class:`gcloud.storage.bucket.Bucket` 

187 :returns: The bucket matching the name provided or None if not found. 

188 """ 

189 try: 

190 return self.get_bucket(bucket_name) 

191 except NotFound: 

192 return None

220 """Get all buckets in the project associated to the client. 

221 

222 This will not populate the list of blobs available in each
